<script>
// FAQ Accordion & Progressive Loading Functionality
document.addEventListener('DOMContentLoaded', function() {
  const faqItems = document.querySelectorAll('.faq-item');
  const categoryButtons = document.querySelectorAll('.category-btn');
  const INITIAL_ITEMS = 6; // Show 6 FAQs initially
  const LOAD_MORE_COUNT = 6; // Load 6 more each time
  let currentlyVisible = INITIAL_ITEMS;

  // Initialize - hide items beyond initial count
  function initializeFAQs() {
    faqItems.forEach((item, index) => {
      if (index >= INITIAL_ITEMS) {
        item.style.display = 'none';
        item.classList.add('hidden-faq');
      }
    });
    updateLoadMoreButton();
  }

  // Accordion functionality
  faqItems.forEach(item => {
    const question = item.querySelector('.faq-question');
    
    question.addEventListener('click', function() {
      const isActive = item.classList.contains('active');
      
      // Close all items
      faqItems.forEach(i => i.classList.remove('active'));
      
      // Open clicked item if it wasn't active
      if (!isActive) {
        item.classList.add('active');
      }
    });
  });

  // Category filter functionality
  categoryButtons.forEach(button => {
    button.addEventListener('click', function() {
      // Remove active class from all buttons
      categoryButtons.forEach(btn => btn.classList.remove('active'));
      
      // Add active class to clicked button
      this.classList.add('active');
      
      // Get filter value
      const filterValue = this.getAttribute('data-category');
      
      // Reset visibility counter
      currentlyVisible = INITIAL_ITEMS;
      
      // Filter FAQ items
      let visibleCount = 0;
      faqItems.forEach(item => {
        const itemCategory = item.getAttribute('data-category');
        
        if (filterValue === 'all' || itemCategory === filterValue) {
          if (visibleCount < INITIAL_ITEMS) {
            item.style.display = 'block';
            item.classList.remove('hidden-faq');
            visibleCount++;
          } else {
            item.style.display = 'none';
            item.classList.add('hidden-faq');
          }
        } else {
          item.style.display = 'none';
          item.classList.add('hidden-faq');
          item.classList.remove('active'); // Close hidden items
        }
      });
      
      updateLoadMoreButton();
    });
  });

  // Load More functionality
  const loadMoreBtn = document.getElementById('faq-load-more-btn');
  if (loadMoreBtn) {
    loadMoreBtn.addEventListener('click', function() {
      const activeCategory = document.querySelector('.category-btn.active').getAttribute('data-category');
      const hiddenItems = Array.from(faqItems).filter(item => {
        const itemCategory = item.getAttribute('data-category');
        const isHidden = item.classList.contains('hidden-faq');
        const matchesCategory = activeCategory === 'all' || itemCategory === activeCategory;
        return isHidden && matchesCategory;
      });

      // Show next batch
      hiddenItems.slice(0, LOAD_MORE_COUNT).forEach((item, index) => {
        setTimeout(() => {
          item.style.display = 'block';
          item.classList.remove('hidden-faq');
          // Trigger animation
          item.style.opacity = '0';
          item.style.transform = 'translateY(20px)';
          setTimeout(() => {
            item.style.transition = 'all 0.5s ease';
            item.style.opacity = '1';
            item.style.transform = 'translateY(0)';
          }, 50);
        }, index * 100);
      });

      currentlyVisible += LOAD_MORE_COUNT;
      
      // Update button after animations
      setTimeout(() => {
        updateLoadMoreButton();
      }, LOAD_MORE_COUNT * 100 + 500);
    });
  }

  // Update Load More button visibility and text
  function updateLoadMoreButton() {
    const activeCategory = document.querySelector('.category-btn.active').getAttribute('data-category');
    const hiddenItems = Array.from(faqItems).filter(item => {
      const itemCategory = item.getAttribute('data-category');
      const isHidden = item.classList.contains('hidden-faq');
      const matchesCategory = activeCategory === 'all' || itemCategory === activeCategory;
      return isHidden && matchesCategory;
    });

    const loadMoreContainer = document.getElementById('faq-load-more');
    const loadMoreBtn = document.getElementById('faq-load-more-btn');
    const remainingCount = document.getElementById('faq-remaining-count');

    if (hiddenItems.length > 0) {
      loadMoreContainer.style.display = 'flex';
      remainingCount.textContent = hiddenItems.length;
    } else {
      loadMoreContainer.style.display = 'none';
    }
  }

  // Initialize on page load
  initializeFAQs();
});
</script>
