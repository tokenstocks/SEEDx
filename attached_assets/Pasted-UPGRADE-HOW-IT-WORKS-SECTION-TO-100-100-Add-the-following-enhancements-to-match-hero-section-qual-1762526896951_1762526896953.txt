JAVASCRIPT FOR TOGGLE FUNCTIONALITY (Add to your script):

<script>
// Audience Toggle Functionality
document.addEventListener('DOMContentLoaded', function() {
  const toggleButtons = document.querySelectorAll('.toggle-btn');
  const primerFlow = document.querySelector('.primer-flow');
  const regeneratorFlow = document.querySelector('.regenerator-flow');

  toggleButtons.forEach(button => {
    button.addEventListener('click', function() {
      // Remove active class from all buttons
      toggleButtons.forEach(btn => btn.classList.remove('active'));
      
      // Add active class to clicked button
      this.classList.add('active');
      
      // Show/hide appropriate flow
      const audience = this.getAttribute('data-audience');
      if (audience === 'primer') {
        primerFlow.classList.add('active-flow');
        regeneratorFlow.classList.remove('active-flow');
      } else {
        regeneratorFlow.classList.add('active-flow');
        primerFlow.classList.remove('active-flow');
      }
    });
  });

  // Video placeholder click handler
  const videoPlaceholder = document.querySelector('.video-placeholder');
  if (videoPlaceholder) {
    videoPlaceholder.addEventListener('click', function() {
      // Replace with actual video embed or modal
      alert('Video player would open here. Replace with your video URL.');
    });
  }
});
</script>
